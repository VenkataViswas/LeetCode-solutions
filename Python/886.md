# 886. Possible Bipartition

## Solution 1 (time O(n+m), space O(n+m))

```python
class Solution(object):
    def possibleBipartition(self, n, dislikes):
        """
        :type n: int
        :type dislikes: List[List[int]]
        :rtype: bool
        """
        g = collections.defaultdict(list)
        for x, y in dislikes:
            g[x - 1].append(y - 1)
            g[y - 1].append(x - 1)
        color = {}

        def dfs(cur_node, cur_color):
            new_color = 1 - cur_color
            flag = True
            for new_node in g[cur_node]:
                if new_node in color:
                    if color[new_node] != new_color:
                        flag = False
                        break
                else:
                    color[new_node] = new_color
                    if not dfs(new_node, new_color):
                        flag = False
                        break
            return flag
        
        for i in range(n):
            if i in color:
                continue
            color[i] = 0
            if not dfs(i, 0):
                return False
        return True
```
